  <script>
    // set year
    document.getElementById('year').textContent = new Date().getFullYear();

    // Lightbox
    const gallery = document.querySelectorAll('.gallery img');
    const lb = document.getElementById('lightbox');
    const lbImg = document.getElementById('lightboxImg');
    gallery.forEach(img=>{
      img.addEventListener('click',()=>{
        lbImg.src = img.dataset.full || img.src;
        lb.classList.add('show');
      })
    });
    function closeLightbox(e){
      if(e.target !== lbImg) lb.classList.remove('show');
    }

    // Form handlers (simulation)
    function handleForm(e){
      e.preventDefault();
      const data = {
        nama:document.getElementById('nama').value,
        usia:document.getElementById('usia').value,
        prodi:document.getElementById('prodi').value,
        kamar:document.getElementById('kamar').value,
        alamat:document.getElementById('alamat').value
      };
      // Simulasi: tampilkan ringkasan pendaftaran
      alert('Pendaftaran terkirim!\n\nNama: '+data.nama+'\nProgram: '+data.prodi+'\nJenis kamar: '+data.kamar);
      e.target.reset();
      return false;
    }

    function handleMsg(e){
      e.preventDefault();
      const nama = document.getElementById('msgNama').value;
      const email = document.getElementById('msgEmail').value;
      alert('Terima kasih, '+nama+'! Pesan Anda telah diterima (simulasi). Kami akan membalas ke '+email);
      e.target.reset();
      return false;
    }
  </script>
